6. Database Design
users
CREATE TABLE users (
  id UUID PRIMARY KEY,
  name TEXT,
  email TEXT UNIQUE,
  image TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
conversations
CREATE TABLE conversations (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  title TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
messages
CREATE TABLE messages (
  id UUID PRIMARY KEY,
  conversation_id UUID REFERENCES conversations(id),
  role TEXT,
  content TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
products_cache
CREATE TABLE products_cache (
  id BIGINT PRIMARY KEY,
  name TEXT,
  slug TEXT,
  category TEXT,
  brand TEXT,
  price NUMERIC,
  stock INTEGER,
  specs JSONB,
  short_description TEXT,
  updated_at TIMESTAMP
);
